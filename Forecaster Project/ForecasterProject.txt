Your app will:
	1.	Monitor a NASDAQ stock

3.3 Feature Engineering

# features/engineering.py
import pandas as pd
import numpy as np

class FeatureEngineer:
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        df["log_close"] = np.log(df["Close"])
        df["returns"] = df["Close"].pct_change()
        df["rolling_mean_14"] = df["Close"].rolling(14).mean()
        df["rolling_std_14"] = df["Close"].rolling(14).std()
        return df.dropna()

# models/base.py
from abc import ABC, abstractmethod
import pandas as pd

class BaseForecastModel(ABC):
    name: str

    @abstractmethod
    def fit(self, train: pd.Series):
        pass

    @abstractmethod
    def predict(self, steps: int):
        pass

3.5 Example: ARIMA Model

# models/arima_model.py
from statsmodels.tsa.arima.model import ARIMA
from .base import BaseForecastModel

class ARIMAModel(BaseForecastModel):
    name = "ARIMA"

    def __init__(self, order=(1,1,1)):
        self.order = order
        self.model = None

    def fit(self, train):
        self.model = ARIMA(train, order=self.order).fit()

    def predict(self, steps: int):
        return self.model.forecast(steps=steps)

3.6 Model Evaluation


# evaluation/metrics.py
import numpy as np

class Metrics:
    @staticmethod
    def rmse(y_true, y_pred):
        return np.sqrt(np.mean((y_true - y_pred) ** 2))

    @staticmethod
    def mae(y_true, y_pred):
        return np.mean(np.abs(y_true - y_pred))

3.7 Model Selector

# evaluation/model_selector.py
import time

class ModelSelector:
    def select_best(self, models, train, test):
        results = []

        for model in models:
            start = time.time()
            model.fit(train)
            preds = model.predict(len(test))
            duration = time.time() - start

            results.append({
                "model": model.name,
                "rmse": Metrics.rmse(test, preds),
                "training_time": duration
            })

        return sorted(results, key=lambda x: x["rmse"])[0]
